18  package com.agorapulse.micronaut.log4aws.http;
19  
20  import io.micronaut.http.HttpRequest;
21  import io.sentry.EventProcessor;
22  import io.sentry.SentryEvent;
23  
24  import java.util.Arrays;
25  import java.util.Collections;
26  import java.util.HashMap;
27  import java.util.List;
28  import java.util.Map;
29  import java.util.stream.Collectors;
30  import java.util.stream.StreamSupport;
31  
32  public class MicronautRequestEventProcessor implements EventProcessor {
33  
34      private static final List<String> SENSITIVE_HEADERS = Arrays.asList("X-FORWARDED-FOR", "AUTHORIZATION", "COOKIE");
35  
36      private static final List<String> SENSITIVE_PARAMS = Collections.singletonList("TOKEN");
37  
38      private final HttpRequest<?> request;
39  
40      public MicronautRequestEventProcessor(HttpRequest<?> request) {
41          this.request = request;
42      }
43  
44      @Override
45      public SentryEvent process(SentryEvent event, Object hint) {
46          event.setTag("req.path", request.getPath());
47          event.setTag("req.method", request.getMethod().toString());
48  
49          // it can be actually null on read timeout
50          if (request.getRemoteAddress() != null) {
51              event.setTag("req.remoteHost", request.getRemoteAddress().getHostString());
52          }
53  
54          event.setTag("req.serverHost", request.getServerAddress().getHostString());
55          event.setExtra("req.parameters", resolveParameters(request));
56          event.setExtra("req.headers", resolveHeaders(request));
57          return event;
58      }
59  
60      private static Map<String, String> resolveParameters(final HttpRequest<?> request) {
61          return StreamSupport.stream(request.getParameters().spliterator(), false)
62              .filter(e -> !SENSITIVE_PARAMS.contains(e.getKey().toUpperCase()))
63              .collect(Collectors.toMap(Map.Entry::getKey, e -> String.join(",", e.getValue())));
64      }
65  
66      private Map<String, String> resolveHeaders(final HttpRequest<?> request) {
67          final Map<String, String> headersMap = new HashMap<>();
68          for (String headerName : request.getHeaders().names()) {
69              if (!SENSITIVE_HEADERS.contains(headerName.toUpperCase())) {
70                  headersMap.put(headerName, String.join(",", request.getHeaders().getAll(headerName)));
71              }
72          }
73          return headersMap;
74      }
75  
76      @Override
77      public String toString() {
78          return String.format("MicronautRequestBuildHelper{ request = %s }", request);
79      }
80  }
